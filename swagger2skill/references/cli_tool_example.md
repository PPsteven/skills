# CLI Tool Example and Best Practices

This document shows example Click-based CLI tool implementations generated by swagger2skill.

## Basic Structure

All generated CLI tools follow this pattern:

```python
#!/usr/bin/env python3
"""
CLI Tool - {Skill Name}

Auto-generated CLI tool for managing {API Type} APIs.
"""

import click
import json
import requests
from typing import Optional
import os


class APIClient:
    """API client wrapper."""

    def __init__(self, base_url: str = None, token: str = None):
        """Initialize API client."""
        self.base_url = base_url or os.getenv('API_BASE_URL', 'http://localhost:8080')
        self.token = token or os.getenv('API_TOKEN', '')
        self.headers = {'Content-Type': 'application/json'}
        if self.token:
            self.headers['Authorization'] = f'Bearer {self.token}'

    def request(self, method: str, endpoint: str, **kwargs) -> dict:
        """Make API request."""
        url = f"{self.base_url}{endpoint}"
        try:
            response = requests.request(method, url, headers=self.headers, **kwargs, timeout=30)
            response.raise_for_status()
            return response.json() if response.text else {"status": "success"}
        except Exception as e:
            click.echo(f"❌ Error: {e}", err=True)
            return {"error": str(e)}


api = APIClient()


@click.group()
def cli():
    """Main CLI Tool"""
    pass


# Category subcommands...

if __name__ == '__main__':
    cli()
```

## Example 1: Simple List/Detail Pattern

```python
@cli.group(name='item')
def item_group():
    """Manage items"""
    pass


@item_group.command('list')
@click.option('--limit', default=10, help='Number of items to return')
@click.option('--offset', default=0, help='Pagination offset')
def item_list(limit, offset):
    """List all items"""
    click.echo(f"📋 Fetching items (limit={limit}, offset={offset})...")
    result = api.request('GET', f'/api/v1/items?limit={limit}&offset={offset}')

    if 'error' in result:
        click.echo(f"❌ {result['error']}", err=True)
        return

    items = result.get('items', [])
    click.echo(f"\n✅ Found {len(items)} items:\n")
    for item in items:
        click.echo(f"  • {item['id']}: {item['name']}")


@item_group.command('detail')
@click.argument('item_id')
def item_detail(item_id):
    """Get item details"""
    click.echo(f"🔍 Fetching details for item {item_id}...")
    result = api.request('GET', f'/api/v1/items/{item_id}')

    if 'error' in result:
        click.echo(f"❌ {result['error']}", err=True)
        return

    click.echo("\n📄 Item Details:\n")
    click.echo(json.dumps(result, indent=2))
```

## Example 2: Create/Update Operations

```python
@item_group.command('create')
@click.option('--name', required=True, help='Item name')
@click.option('--description', default='', help='Item description')
@click.option('--tags', default='', help='Comma-separated tags')
def item_create(name, description, tags):
    """Create a new item"""
    click.echo(f"➕ Creating item: {name}...")

    payload = {
        'name': name,
        'description': description,
    }
    if tags:
        payload['tags'] = tags.split(',')

    result = api.request('POST', '/api/v1/items', json=payload)

    if 'error' in result:
        click.echo(f"❌ {result['error']}", err=True)
        return

    click.echo(f"✅ Item created successfully!")
    click.echo(json.dumps(result, indent=2))


@item_group.command('update')
@click.argument('item_id')
@click.option('--name', help='New name')
@click.option('--description', help='New description')
@click.option('--tags', help='Comma-separated tags')
def item_update(item_id, name, description, tags):
    """Update an existing item"""
    click.echo(f"✏️  Updating item {item_id}...")

    payload = {}
    if name:
        payload['name'] = name
    if description:
        payload['description'] = description
    if tags:
        payload['tags'] = tags.split(',')

    result = api.request('PUT', f'/api/v1/items/{item_id}', json=payload)

    if 'error' in result:
        click.echo(f"❌ {result['error']}", err=True)
        return

    click.echo(f"✅ Item updated successfully!")
```

## Example 3: Complex Query Parameters

```python
@item_group.command('search')
@click.option('--query', required=True, help='Search query')
@click.option('--type', help='Item type filter')
@click.option('--status', help='Status filter')
@click.option('--sort', default='created', help='Sort field')
@click.option('--order', default='desc', help='Sort order (asc/desc)')
@click.option('--limit', default=20, help='Result limit')
def item_search(query, type, status, sort, order, limit):
    """Search for items"""
    click.echo(f"🔎 Searching for: {query}...")

    params = {
        'q': query,
        'limit': limit,
        'sort': f"{sort}:{order}",
    }
    if type:
        params['type'] = type
    if status:
        params['status'] = status

    # Build query string manually
    query_string = '&'.join(f"{k}={v}" for k, v in params.items())
    result = api.request('GET', f'/api/v1/items/search?{query_string}')

    if 'error' in result:
        click.echo(f"❌ {result['error']}", err=True)
        return

    items = result.get('items', [])
    click.echo(f"\n✅ Found {len(items)} results:\n")
    for item in items:
        click.echo(f"  • {item['id']}: {item['name']} ({item.get('status', 'unknown')})")
```

## Example 4: JSON Input/Output

```python
@item_group.command('batch-create')
@click.argument('json_file', type=click.File('r'))
def item_batch_create(json_file):
    """Create multiple items from JSON file"""
    try:
        items = json.load(json_file)
        if not isinstance(items, list):
            click.echo("❌ JSON must be an array of items", err=True)
            return

        click.echo(f"📦 Creating {len(items)} items...")
        success = 0
        failed = 0

        for item in items:
            result = api.request('POST', '/api/v1/items', json=item)
            if 'error' not in result:
                success += 1
                click.echo(f"  ✓ {item['name']}")
            else:
                failed += 1
                click.echo(f"  ✗ {item['name']}: {result['error']}", err=True)

        click.echo(f"\n📊 Results: {success} created, {failed} failed")

    except json.JSONDecodeError:
        click.echo("❌ Invalid JSON file", err=True)


@item_group.command('export')
@click.option('--format', default='json', help='Export format (json/csv)')
@click.argument('output_file', type=click.File('w'))
def item_export(format, output_file):
    """Export all items"""
    click.echo("📤 Exporting items...")

    result = api.request('GET', '/api/v1/items?limit=1000')

    if 'error' in result:
        click.echo(f"❌ {result['error']}", err=True)
        return

    items = result.get('items', [])

    if format == 'json':
        json.dump(items, output_file, indent=2)
        click.echo(f"✅ Exported {len(items)} items to JSON")
    elif format == 'csv':
        import csv
        if items:
            writer = csv.DictWriter(output_file, fieldnames=items[0].keys())
            writer.writeheader()
            writer.writerows(items)
            click.echo(f"✅ Exported {len(items)} items to CSV")
```

## Example 5: Error Handling and Validation

```python
@item_group.command('delete')
@click.argument('item_id')
@click.option('--force', is_flag=True, help='Skip confirmation')
def item_delete(item_id, force):
    """Delete an item"""
    if not force:
        if not click.confirm(f"🗑️  Are you sure you want to delete {item_id}?"):
            click.echo("Cancelled.")
            return

    click.echo(f"🗑️  Deleting item {item_id}...")
    result = api.request('DELETE', f'/api/v1/items/{item_id}')

    if 'error' in result:
        if 'Not found' in result['error']:
            click.echo(f"❌ Item {item_id} not found", err=True)
        elif 'Forbidden' in result['error']:
            click.echo(f"❌ Permission denied", err=True)
        else:
            click.echo(f"❌ {result['error']}", err=True)
        return

    click.echo(f"✅ Item deleted successfully!")
```

## Best Practices for Generated CLI Tools

### 1. **Environment Variables for Configuration**
```python
base_url = os.getenv('API_BASE_URL', 'http://localhost:8080')
token = os.getenv('API_TOKEN', '')
```

### 2. **Clear Feedback with Emojis**
```python
click.echo("📋 Fetching...")      # Loading
click.echo("✅ Success!")          # Success
click.echo("❌ Error")             # Error
click.echo("⚠️  Warning")          # Warning
click.echo("📤 Exporting...")      # Action
```

### 3. **Proper Error Handling**
```python
try:
    response = requests.get(url, timeout=30)
    response.raise_for_status()
    return response.json()
except requests.Timeout:
    click.echo("❌ Request timeout", err=True)
except requests.HTTPError as e:
    click.echo(f"❌ HTTP {e.response.status_code}: {e.response.text}", err=True)
except Exception as e:
    click.echo(f"❌ Error: {e}", err=True)
```

### 4. **JSON Output for Parsing**
```python
@click.option('--json', is_flag=True, help='Output as JSON')
def command(json):
    """Command with JSON output option."""
    result = api.request('GET', '/api/v1/data')
    if json:
        click.echo(json.dumps(result, indent=2))
    else:
        # Pretty print for humans
        for item in result.get('items', []):
            click.echo(f"  • {item['id']}: {item['name']}")
```

### 5. **Pagination Support**
```python
@click.option('--limit', default=20)
@click.option('--offset', default=0)
def list_command(limit, offset):
    """List with pagination."""
    result = api.request('GET', f'/api/v1/items?limit={limit}&offset={offset}')
```

### 6. **Required vs Optional Parameters**
```python
@click.argument('item_id')                    # Required
@click.option('--name', required=True)        # Required option
@click.option('--description', default='')   # Optional with default
@click.option('--tags')                       # Optional
```

- `requests` - HTTP requests
- `json` - JSON handling (built-in)

## Usage in Generated Skills

After generation, users extend the CLI tool by:

1. **Implementing TODO endpoints** - Replace `# TODO: Implement` markers with actual API calls
2. **Adding error handling** - Wrap requests with try-except blocks
3. **Adding authentication** - Configure API token/credentials
4. **Testing locally** - Run CLI commands before integration
5. **Documenting** - Update SKILL.md with actual usage examples
